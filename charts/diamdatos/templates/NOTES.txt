1. Get the application address by running these commands:
{{- if .Values.diam3gpp.enabled }}
DIAM3GPP:
========
  > export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "aachart.fullname" . }}-{{ lower .Values.diam3gpp.funcionalidad }})
  > export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  > echo $NODE_IP:$NODE_PORT
  and connect Gy diameter client to $NODE_IP:$NODE_PORT
{{- end }}
